def read_json_file(filename):
    
    data = []
    with open(filename, "r") as file:
        for line in file:
            data.append(json.loads(line))
            
    return data

df = pd.DataFrame()

ind_counter = 1
for ind in pop_size_12:

    file_name = ""

    for f in os.listdir(f"{ga_metrics_folder}/{ind}/"):
        if "metrics" in f:
            print(f"Metrics file found: {f}")
            file_name = f

    if file_name == "":
        print(f"No metrics file found for individual {ind}.")
    else:
        json_data = read_json_file(f"{ga_metrics_folder}/{ind}/{file_name}")

        x = [d["iteration"] for d in json_data]
        y = [d["best_fitness"] for d in json_data]
        df_tmp = pd.DataFrame({'generation': x, 'fitness': y, 'individual': f"ind {ind_counter}"})

        df = pd.concat([df, df_tmp])

    ind_counter += 1
